一、Virtual DOM

1.1 Virtual DOM 是什么?

利用普通的javaScript对象描述真实DOM,这个javaScript对象就是虚拟DOM

1.2 为什么要使用Virtual DOM

  • DOM操作比较复杂
  • 需要兼容各种版本的浏览器
  • 每进行一次DOM操作,就会重述和回流,性能不好

1.3 Virtual DOM的作用

  • 维护视图和状态的关系
  • 对于复杂的视图,可以减少DOM操作
  • 可以实现服务端渲染SSR、原生应用、小程序登等

1.4 Virtual DOM的库

二、snabbdom的基本使用

2.1 实现一个 hello world

/**
 * h 但会一个虚拟dom
 * init 是一个高阶函数  返回patch---对比两个vnode的差异更新到真实dom
 * thunk 是一种优化策略
 */
import { h } from 'snabbdom/build/package/h'
import { init } from 'snabbdom/build/package/init'

// 1. 实现hello world
// 初始化patch  对比两个vnode的差异更新到真实dom
const patch = init([])
/**
 * h 函数
 * 第一个参数 标签+选择器
 * 第二个参数 字符串代表标签的内容
 */
let vnode = h('div#contanier.cls', 'hello world')

let app = document.querySelector('#app')
/**
 * patch  两个不同时,后者会替换前者
 * 第一个参数 可以是真实的dom元素
 * 第二个参数 vnode
 */
let oldVnode = patch(app, vnode)

vnode = h('div', 'hello snabbdomh')

patch(oldVnode, vnode)

2.2 实现 div内嵌套 div和p

import { h } from 'snabbdom/build/package/h'
import { init } from 'snabbdom/build/package/init'

let patch = init([])

let vnode = h('div#container', [
  h('h1', 'hello H1'),
  h('p', '这是一个块元素')
])

let app = document.querySelector('#app')

let oldVnode = patch(app, vnode)

// 延时更细 vnode
setTimeout(() => {
  vnode = h('div#container', [
    h('h1', 'hello 8888888'),
    h('p', '这是一个更新的块元素')
  ])
  patch(oldVnode, vnode)
},3000)

2.3 清空元素

// 官方使用 null ----是错误的  Cannot read property 'key' of null
patch(oldVnode, null)

// 实际解决方案 h('!') 注释节点
patch(oldVnode, h('!'))

2.4 模块使用

import { h } from 'snabbdom/build/package/h'
import { init } from 'snabbdom/build/package/init'
import { styleModule }  from 'snabbdom/build/package/modules/style'
import { eventListenersModule }  from 'snabbdom/build/package/modules/eventlisteners'

// 初始化时,加载模块
let patch = init([
  styleModule,
  eventListenersModule
])

// 第二个参数是对象时, 添加的属性
let vnode = h('div#container',{
  style: {
    backgroundColor: 'red',
    padding:'10px'
  },
  on: {
    click: eventHandler
  }
},[
  h('h1','hello world'),
  h('p', '这是一个p标签')
])

function eventHandler (){
  console.log('点击了div')
}

let app = document.querySelector('#app')

patch(app, vnode)

三、snabbdom的源码解析

  • h() 函数创建js对象描述真实DOM
  • init 初始化模块+添加style、eventlistener,并创建patch
  • patch 比较新旧两个vnode,并将其的变化更新到真实的DOM上

3.1 h函数

import { vnode, VNode, VNodeData } from './vnode'
import * as is from './is'

export type VNodes = VNode[]
export type VNodeChildElement = VNode | string | number | undefined | null
export type ArrayOrElement<T> = T | T[]
export type VNodeChildren = ArrayOrElement<VNodeChildElement>

function addNS(data: any, children: VNodes | undefined, sel: string | undefined): void {
  data.ns = 'http://www.w3.org/2000/svg'
  if (sel !== 'foreignObject' && children !== undefined) {
    for (let i = 0; i < children.length; ++i) {
      const childData = children[i].data
      if (childData !== undefined) {
        addNS(childData, (children[i] as VNode).children as VNodes, children[i].sel)
      }
    }
  }
}

// h 函数的重载
export function h(sel: string): VNode
export function h(sel: string, data: VNodeData | null): VNode
export function h(sel: string, children: VNodeChildren): VNode
export function h(sel: string, data: VNodeData | null, children: VNodeChildren): VNode
export function h(sel: any, b?: any, c?: any): VNode {
  var data: VNodeData = {}
  var children: any
  var text: any
  var i: number
  // 说明穿了 三个参数
  if (c !== undefined) {
    // b 不为空时,data 是vnode的属性和状态
    if (b !== null) { data = b }
    // c 是子元素
    if (is.array(c)) {
      children = c
    // c是文本或者是数字 
    } else if (is.primitive(c)) {
      text = c
    // c 是一个vnode
    } else if (c && c.sel) {
      children = [c]
    }
  // 两个参数
  } else if (b !== undefined && b !== null) {
    if (is.array(b)) {
      children = b
    } else if (is.primitive(b)) {
      text = b
    } else if (b && b.sel) {
      children = [b]
    } else { data = b }
  }
  // 如果children 存在,遍历children 并创建vnode节点
  if (children !== undefined) {
    for (i = 0; i < children.length; ++i) {
      // 判断 children是不是 string/number,  创建文本节点
      if (is.primitive(children[i])) children[i] = vnode(undefined, undefined, undefined, children[i], undefined)
    }
  }
  // 判断是不是 svg
  if (
    sel[0] === 's' && sel[1] === 'v' && sel[2] === 'g' &&
    (sel.length === 3 || sel[3] === '.' || sel[3] === '#')
  ) {
    // 如果是 svg 添加命名空间
    addNS(data, children, sel)
  }
  // 返回 vnode 创建虚拟节点
  return vnode(sel, data, children, text, undefined)
};

    // VNode 接口 约束所有使用的对象都拥有这些属性
export interface VNode {
  // 标签+选择器
  sel: string | undefined
  // 样式/事件/属性
  data: VNodeData | undefined
  // 子节点 和text 文本节点 只能有一个
  children: Array<VNode | string> | undefined
  // 保存 vnode最后生成的真实dom
  elm: Node | undefined
  // 文本节点
  text: string | undefined
  // 优化
  key: Key | undefined
}

// 定义属性/样式/事件 的接口
export interface VNodeData {
  props?: Props
  attrs?: Attrs
  class?: Classes
  style?: VNodeStyle
  dataset?: Dataset
  on?: On
  hero?: Hero
  attachData?: AttachData
  hook?: Hooks
  key?: Key
  ns?: string // for SVGs
  fn?: () => VNode // for thunks
  args?: any[] // for thunks
  [key: string]: any // for any other 3rd party module
}

export function vnode (sel: string | undefined,
  data: any | undefined,
  children: Array<VNode | string> | undefined,
  text: string | undefined,
  elm: Element | Text | undefined): VNode {
  // 如果 data中有key,将设置 key的值
  const key = data === undefined ? undefined : data.key
  return { sel, data, children, text, elm, key }
}
